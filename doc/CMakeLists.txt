set(OPENSSL_DOCS_FULLLIST)
macro(perldoc_add_document_man PODFILE_OR_IN DEFAULTPAGENUMBER)
    set(_PAGENUMBER ${DEFAULTPAGENUMBER})
    get_filename_component(_PODFILE_OR_IN_LAST_EXT "${PODFILE_OR_IN}" LAST_EXT)
    if (_PODFILE_OR_IN_LAST_EXT STREQUAL ".in")
        get_filename_component(_PODFILE_FILE_NAME "${PODFILE_OR_IN}" NAME_WLE)
        set(_PODFILE "${CMAKE_CURRENT_BINARY_DIR}/pod/man${_PAGENUMBER}/${_PODFILE_FILE_NAME}")
    else()
        set(_PODFILE ${PODFILE_OR_IN})
    endif()

    get_filename_component(_PODFILE_OR_IN_ABSOLUTE ${PODFILE_OR_IN} ABSOLUTE)
    get_filename_component(_PODFILE_ABSOLUTE ${_PODFILE} ABSOLUTE)
    get_filename_component(_MAINPAGENAME ${_PODFILE} NAME_WLE)
    execute_process(COMMAND ${PERL_EXECUTABLE} ${CMAKE_CURRENT_SOURCE_DIR}/extract-names.pl
        INPUT_FILE ${_PODFILE_OR_IN_ABSOLUTE}
        OUTPUT_VARIABLE _PAGENAME_OUTPUT
        COMMAND_ERROR_IS_FATAL ANY
    )
    string(STRIP ${_PAGENAME_OUTPUT} _PAGENAMES)
    string(REGEX REPLACE "[ \t\r\n]+" ";" _PAGENAMELIST ${_PAGENAMES})
    set(_PAGEOUTPUTLIST)
    foreach (_PAGENAME IN LISTS _PAGENAMELIST)
        if ( NOT "${_PAGENAME}" STREQUAL "${_MAINPAGENAME}" )
            list(FIND OPENSSL_DOCS_FULLLIST "man/man${_PAGENUMBER}/${_PAGENAME}.${_PAGENUMBER}" _FULLLISTINDEX)
            if (NOT _FULLLISTINDEX EQUAL -1)
                set(_PAGEOUTPUTLIST ${_PAGEOUTPUTLIST} "man/man${_PAGENUMBER}/${_PAGENAME}.${_PAGENUMBER}")
            endif()
            unset(_FULLLISTINDEX)
        endif()
    endforeach()
    if (_PODFILE_OR_IN_LAST_EXT STREQUAL ".in")
        # N.B. DO EDIT generatepod.cmake WHEN OpenSSL UPDATES!
        add_custom_command(OUTPUT ${_PODFILE}
                           COMMAND ${CMAKE_COMMAND}
                                   "-DOPENSSL_DEPRECATED_3_0:BOOL=${OPENSSL_DEPRECATED_3_0}"
                                   "-DOPENSSL_MIN_API=${OPENSSL_MIN_API}"
                                   "-DINPUTFILEPATH=${PODFILE_OR_IN}"
                                   "-DOUTPUTFILEPATH=${_PODFILE}"
                                   -P ${CMAKE_CURRENT_LIST_DIR}/generatepod.cmake
                           MAIN_DEPENDENCY ${PODFILE_OR_IN}
                           DEPENDS generatepod.cmake
        )
    endif()
    add_custom_command(OUTPUT "man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}"
                       BYPRODUCTS ${_PAGEOUTPUTLIST}
                       COMMAND ${CMAKE_COMMAND}
                               "-DSECTION=${_PAGENUMBER}"
                               "-DPODFILEPATH=${_PODFILE_ABSOLUTE}"
                               "-DLINKS=\"${_PAGEOUTPUTLIST}\""
                               "-DOUTPUTFILEPATH=man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}"
                               "-P" generateDoc.cmake
                       MAIN_DEPENDENCY ${_PODFILE}
                       DEPENDS generateDoc.cmake
    )
    set(OPENSSL_DOCS_LIST ${OPENSSL_DOCS_LIST} "man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}")
    install(FILES ${CMAKE_CURRENT_BINARY_DIR}/man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}
        DESTINATION ${CMAKE_INSTALL_MANDIR}/man${_PAGENUMBER}
        RENAME ${_MAINPAGENAME}.${_PAGENUMBER}ossl
    )
    foreach (MANFILE IN LISTS _PAGENAMELIST)
        install(FILES ${CMAKE_CURRENT_BINARY_DIR}/man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}
            DESTINATION ${CMAKE_INSTALL_MANDIR}/man${_PAGENUMBER}
            RENAME ${MANFILE}.${_PAGENUMBER}ossl
        )
    endforeach()
    set(OPENSSL_DOCS_FULLLIST ${OPENSSL_DOCS_FULLLIST} ${_PAGEOUTPUTLIST} "man/man${_PAGENUMBER}/${_MAINPAGENAME}.${_PAGENUMBER}")
    unset(_REPLACED_OUTPUTNAME)
    unset(_PAGEOUTPUTLIST)
    unset(_PAGENAMELIST)
    unset(_PAGENAMES)
    unset(_PAGENAME_OUTPUT)
    unset(_PAGENUMBER)
    unset(_PAGENUMBER_OUTPUT)
    unset(_MAINPAGENAME)
    unset(_PODFILE_ABSOLUTE)
    unset(_PODFILE_OR_IN_ABSOLUTE)
    unset(_PODFILE)
endmacro()
